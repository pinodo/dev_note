# Create a new table
```shell
class MEMBER
{
  ID        int,    ->  VARCHAR2(50)
  PWD       String, ->  VARCHAR2(50)
  NAME      String, ->  VARCHAR2(50)
  GENDER    String, ->  VARCHAR2(50)
  AGE       int,    ->  NUMBER
  BIRTHDAY  String, ->  VARCHAR2(50)
  PHONE     String, ->  VARCHAR2(50)
  REGDATE   Date    ->  DATE
}
```
